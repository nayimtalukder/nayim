function checkBoxes(){
const triggerBotton = window.innerHeight / 5 * 4
boxes.forEach(box => {
const boxTop = box.getBoundingClientRect().top
if(boxTop < triggerBotton){
box.ClassList.add('show')
} else{ box.ClassList.remove('show') } }); }
